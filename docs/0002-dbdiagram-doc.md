// Represents a physical room.
Table room {
  Id bigint [primary key]
  Name nvarchar(100) [not null]
  CreatedAtUtc datetime2(3) [not null]
}

// Represents one ESP32 monitoring unit.
Table monitor {
  Id bigint [primary key]
  RoomId bigint [not null]
  Name nvarchar(100) [not null]
  FirmwareVersion varchar(20) [note: 'Generated by firmware.']
  LastHeartBeat datetime2(3) [null]
  CreatedAtUtc datetime2(3) [not null]
}

// Represents a physical monitored outlet.
Table channel {
  Id bigint [primary key]
  MonitorId bigint [not null]
  PhysicalPort tinyint [not null]
  CreatedAtUtc datetime2(3) [not null]
  Indexes {
    (MonitorId, PhysicalPort) [unique]
  }
}

// Stores the historical identity of what was connected.
Table channelAssignment {
  Id bigint [primary key]
  ChannelId bigint [not null]
  Label nvarchar(100) [not null]
  /*
    A channel may have at most one active assignment 
    (EndDateUtc IS NULL) at any given time. This rule 
    is enforced by the Energy Analytics API.
  */
  StartDateUtc datetime2 [not null]
  EndDateUtc datetime2 [null]
}

/* 
** This is the most important table.
** This is your permanent historical dataset.
*/ 
Table telemetrySample { 
  Id bigint [primary key]
  ChannelId bigint [not null]
  TimeStampUtc datetime2(3) [not null]
  Voltage decimal(8,3) [not null, note: 'RMS Electric Voltage']
  Current decimal(8,3) [not null, note: 'RMS Electric Current']
  Power decimal(10,3) [not null, note: 'Instantaneous Power (W)']
  CreatedAtUtc datetime2(3) [note: 'Ingestion timestamp']
  Indexes {
    (ChannelId, TimeStampUtc)
  }
}

/*
** Analytics projection.
** Generated by the Telemetry Ingestion Service.
*/
Table telemetryMinute {
  ChannelId bigint [primary key] // it will be used as a composite PK with MinuteUtc
  MinuteUtc datetime2(0) [primary key] // it will be used as a composite PK with ChannelId
  SampleCount smallint [not null]
  MinPower decimal(10,3) [not null]
  MaxPower decimal(10,3) [not null]
  AvgPower decimal(10,3) [not null]
  EnergyConsumedWh decimal(12,4) [not null] 
}

// Analytics projection.
Table telemetryHourly {
  ChannelId bigint [primary key] // it will be used as a composite PK with HourUtc
  HourUtc datetime2(0) [primary key] // it will be used as a composite PK with ChannelId
  SampleCount int [not null]
  MinPower decimal(10,3) [not null]
  MaxPower decimal(10,3) [not null] 
  AvgPower decimal(10,3) [not null] 
  EnergyConsumedWh decimal(12,4) [not null] 
}

// Business reporting projection.
Table telemetryDaily {
  ChannelId bigint [primary key] // it will be used as a composite PK with DateUtc
  DateUtc date [primary key] // it will be used as a composite PK with ChannelId
  SampleCount int [not null]
  AvgPowerW decimal(10,3) [not null] 
  PeakPowerW decimal(10,3) [not null] 
  EnergyConsumedWh decimal(12,4) [not null] 
  EstimatedCost decimal(12,4) [not null] 
}

// Required for cost calculations.
Table energyTariff {
  Id bigint [primary key]
  Name nvarchar(100) [not null] 
  CostPerKWh decimal(10,4) [not null] 
  EffectiveFromUtc datetime2(3) [not null]
  EffectiveToUtc datetime2(3) [null]
}

/*
** Tracks the last successfully processed timestamp for each aggregation job
** (minute, hourly, daily). Used by the Telemetry Ingestion Service to resume
** processing after restarts and prevent reprocessing historical telemetry.
*/
Table aggregationCheckpoint
{
  AggregationName varchar(50) [primary key]
  LastProcessedUtc datetime2(3) [not null]
  UpdatedAtUtc datetime2(3) [not null]
}
